function startMove(element,obj,speed,fn){

clearInterval(element.timer)

element.timer = setInterval(function(){

//计时器每次刷新isComplete状态

var isComplete = true;

//通过for in 获得对象中的属性 和 属性值

for(var attr in obj){

//获取 当前页面内目标的样式 并通过计时器实时刷新 调用getStyle方法

var current = Math.round(parseFloat(getStyle(element,attr)));

//如果取不到或者 取到返回的不为num类型 定义为0

if(!current){

current = 0;

}

//减速运动框架

// var speed = (obj[attr]-current)/10;

// if(speed<0){

// speed = Math.floor(speed);

// }else if(speed>0){

// speed = Math.ceil(speed);

// }

//判断运动是否结束

if(obj[attr]==current){


}else{

element.style[attr]=current+speed+”px”;

isComplete = false;

}

}

//运动结束 清空计时器

if(isComplete){

clearInterval(element.timer);

if(fn){

fn();

}

}

},20);

}


//获得内敛样式

function getStyle(element,attr){

if(element.currentStyle){

return element.currentStyle[attr]; //IE

}else{

return window.getComputedStyle(element,false)[attr];//非IE

}

}
